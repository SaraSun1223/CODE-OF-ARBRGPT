In [32]: x = torch.randn(3, 3)

In [33]: x.align_to('N', 'C')

    525
    526             >>> tensor = torch.randn(2, 3, 5, 7, 11)

# Expected behavior

It should really say "Found unnamed dim at index 0 of Tensor[None, None]".
