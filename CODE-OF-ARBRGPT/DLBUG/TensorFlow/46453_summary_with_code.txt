**Standalone code to reproduce the issue**

import os

import tensorflow as tf

## Download and extract the zip

    precision_mode='FP32',

    conversion_params=params)

    for i in range(steps):
        yield (tf.random.uniform([640, 640, 3]), tf.constant(1, dtype=tf.int32))
